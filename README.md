## Install

```bash
npm install --save pin-input-react
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'pin-input-react'
import 'pin-input-react/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
